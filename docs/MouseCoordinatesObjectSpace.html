    Attach a listener to the mousemove event to update the mouse position in object space

      trackMouse() {
    document.addEventListener("mousemove", (event) => {
      this.mouseX = event.clientX - this.ball.x;
      this.mouseY = this.ball.y - event.clientY;
    });
  }

      measureTextWidth(text, font) {

    ctx.font = font;
